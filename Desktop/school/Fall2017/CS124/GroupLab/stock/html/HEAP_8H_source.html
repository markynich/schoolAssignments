144 template <class SOMETYPE>
145 STATUS HEAP<SOMETYPE>::Insert(SOMETYPE x)
146 
147 // Inserts x into the heap.
148 // Returns OK if successful, FAILED if the heap is full.
149 
150 {
151  if (IsFull()) return FAILED;
152 
153  nNodes++; // The last node of the heap is now vacant.
154 
155  // Starting from the (vacant) last node, go from node i to
156  // its parent iParent and,
157 
158  // if it is a max heap then as long as the parent is smaller
159  // than x, move the parent down:
160  // if it is a min heap then as long as the parent is
161  // larger than x, move the x up:
162 
163  int i = nNodes;
164  int iParent;
165  while (i > 1) {
166  iParent = i/2;
167 
168  if(t == MAX)
169  {
170  if (x <= buf[iParent]) break;
171  }
172  else // must be MIN
173  {
174  if( x >= buf[iParent]) break;
175  }
176  buf[i] = buf[iParent];
177  i = iParent;
178  }
179 
180  // Insert x into the created vacancy:
181  buf[i] = x;
182 
183  return OK;
184 }

194 STATUS HEAP<SOMETYPE>::Remove(SOMETYPE &x)
195 
196 // Removes the largest element (from the root of the heap).
197 // Returns OK if successful, FAILED if the heap is empty.
198 
199 {
200  if (IsEmpty()) return FAILED;
201 
202  // Retrieve the top element:
203 
204  x = buf[1];
205 
206  // Starting from the vacant root, go from node iParent to its
207  // larger child i and, as long as that child
208  // is greater than the last element of the heap,
209  // move that child up:
210 
211  int iParent = 1; // root
212  int i = 2; // its left child
213  while (i <= nNodes) {
214  // Set i to the right child, i+1, if it
215  // exists and is larger:
216  if (i < nNodes && buf[i] < buf[i+1]) i++;
217  if (t == MIN)
218  if(i < nNodes && buf[i] > buf[i+1]) i++;
219  if (t == MAX)
220  if(i < nNodes && buf[i] < buf[i+1]) i++;
221  // Compare with the last node:
222  if (t == MAX && buf[i] <= buf[nNodes]) break;
223  if (t == MIN && buf[i] >= buf[nNodes]) break;
224  buf[iParent] = buf[i]; // Move the child up;
225  iParent = i; // buf[iParent] is now vacant.
226  i *= 2; // i is set to its left child
227  }
228 
229  // Move the last element into the created vacancy:
230  if (nNodes > 1) buf[iParent] = buf[nNodes];
231  nNodes--;
232 
233  return OK;
234 }
